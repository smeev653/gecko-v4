	start = 284 #Crimea
	date = 1532.01.01
	time = { months = 60 }
	build_cost = 0
	can_be_moved = no
	starting_tier = 0
	type = monument
	build_trigger = { owner = { has_country_flag = X_CRI_mission_2_completed_cflag } }
	on_built = {
		if = {
			limit = { has_building = cathedral }
			add_base_tax = 3
		}
		else_if = {
			limit = { has_building = temple }
			hidden_effect = { remove_building = temple }
			add_building = cathedral

		owner = { 
			add_mandate = 5
		}

		on_upgraded = { }
	}
	tier_1 = {
		upgrade_time = { months = 120 }
		cost_to_upgrade = { factor = 350 }
		province_modifiers = { }
		area_modifier = {
			local_development_cost = -0.05
			trade_value_modifier = 0.1
			local_tax_modifier = 0.2
		}
		country_modifiers = {
			church_loyalty_modifier = 0.025
			imperial_mandate = 0.05
			meritocracy = 0.1
		}
		on_upgraded = {
			owner = { add_mandate = 10 }
		}
