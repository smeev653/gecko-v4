X_has_latest_building_trigger = {
	[[trade]
		OR = {
			AND = {
				$builder$ = { dip_tech = 22 }
				has_building = stock_exchange
			}
			AND = {
				$builder$ = { dip_tech = 17 }
				has_building = trade_depot
			}
			AND = {
				$builder$ = { NOT = { dip_tech = 17 } }
				has_building = marketplace
			}

	]

			AND = {
