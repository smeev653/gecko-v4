can_make_state = {
	condition = {
		tooltip = "gecko_state"
			potential = {
				FROM = {
						ai = no
				}	
			}
			allow = {
				OR = {
					area_for_scope_province = {
							type = all
							OR = {
								owned_by = root
								owner = {
									is_subject_of = root
								}
							}			
					}
					area_for_scope_province = {
							is_claim = ROOT
					}
				}	
			}
		}
}
