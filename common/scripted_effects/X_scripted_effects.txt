X_full_state_prosperity_effect = {
	if = {
		limit = {
			NOT = {
				area_for_scope_province = {
					type = all
					owned_by = PREV
				}
			}
		}
		area = {
			type = all
			add_permanent_province_modifier = {
				name = X_full_state_prosperity_pmod
				duration = -1
				desc = X_full_state_prosperity_tt
			}
		}
	}
	else = {
		area = {
			type = all
			remove_province_modifier = X_full_state_prosperity_pmod
		}
	}
}
