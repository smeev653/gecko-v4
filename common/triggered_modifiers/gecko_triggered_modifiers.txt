free_colonist = {
	potential = {
			NOT = {
				capital_scope = {
					continent = north_america
				}
			}
			NOT = {
				capital_scope = {
					continent = south_america
				}
			}
			dip_tech = 7	
		}

	trigger = {
		
		NOT = {
			OR = {
				capital_scope = { region = australia_region }
				capital_scope = { region = oceanea_region }
			}
		}
		has_institution = new_world_i
		was_player = yes
	}
	colonists = 2
}

	potential = {
			was_player = yes
			NOT = {total_development = 199}
		}
	

	trigger = {
		
		NOT = {total_development = 199}
		was_player = yes
		NOT = { is_emperor = yes }
		NOT = {tag = MOS }
		NOT = { is_colonial_nation = yes }	
	}
	advisor_cost = -0.15
	advisor_pool = 1
	ae_impact = -0.15
	interest = -0.5
}

To_the_last_man = {
	potential = {
			was_player = yes
			tag = TUR
		}
	

	trigger = {
		NOT = { is_year = 1500 }
		tag = TUR
		OR = {
			is_in_war = {
				defender_leader = TUR
				attackers = MAM
				attackers = HUN
			}
			AND = {
				is_in_war = {
					defender_leader = TUR
					attackers = MAM
				}
				is_in_war = {
					defender_leader = TUR
					attackers = HUN
				}	
			}	
		}
	}	
	defensiveness = 0.2
	movement_speed = 0.2
}
