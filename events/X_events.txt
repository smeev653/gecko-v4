	desc = X_VEN_events.3.d1
	picture = DIPLOMACY_eventPicture
	is_triggered_only = yes
	fire_only_once = yes

			random_country = {
				limit = {
					owns_or_subject_of = 321
					NOT = { tag = ROOT }
					NOT = { tag = CYP }
				}
				save_event_target_as = X_previous_cyprus_controller_event_target
			}
		}
	}
	
	option = { #At long last
		name = X_VEN_events.3.a
		321 = {
			cede_province = ROOT
			add_core = ROOT
			X_center_of_trade_effect = yes
		}
		if = {
			limit = {
				event_target:X_previous_cyprus_controller_event_target = {
					NOT = { tag = ROOT }
					NOT = { tag = CYP }
				}
			}
			event_target:X_previous_cyprus_controller_event_target = {
				country_event = { id = X_VEN_events.4 days = 30 }
			}
		}
	}
	
	option = { #Now is not the time
		name = X_VEN_events.3.b
		add_prestige = -5
	}
}

#Venice Lays Claim to Cyprus
country_event = {
	id = X_VEN_events.4
	title = X_VEN_events.4.t
	desc = X_VEN_events.4.d1
	picture = DIPLOMACY_eventPicture
	is_triggered_only = yes
	fire_only_once = yes
	major = yes 
	major_trigger = {
		OR = {
			tag = FROM
			tag = ROOT
		}
	}
	
	option = { #An unacceptable offense
		ai_chance = { factor = 0 }
		name = X_VEN_events.4.a
		if = {
			limit = { is_core = 321 }
			declare_war_with_cb = {
				who = FROM
				casus_belli = cb_core
				war_goal_province = 321
			}
		}
		else_if = {
			limit = {
				321 = { 
					OR = {
						is_permanent_claim = ROOT
						is_claim = ROOT
					}
				}
			}
			declare_war_with_cb = {
				who = FROM
				casus_belli = cb_conquest
				war_goal_province = 321
			}
		}
		else = {
			321 = { add_claim = ROOT }
			declare_war_with_cb = {
				who = FROM
				casus_belli = cb_conquest
				war_goal_province = 321
			}
		}
	}
	
	option = { #Peace is the best option
		ai_chance = { factor = 100 }
		name = X_VEN_events.4.b
		FROM = { add_truce_with = ROOT }
		add_truce_with = FROM
		add_prestige = -5
	}	
}
