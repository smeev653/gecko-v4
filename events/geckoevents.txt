				limit = {
					is_state = yes
					is_capital = no
				}
				save_event_target_as = gecko_pol_nobles2
			}
		}
	 }

		months = 12
	}

	option = {
		name = gecko.1.a
		add_estate_loyalty = {
			estate = estate_nobles
			loyalty = -20
		}
		define_ruler = {
			culture = ROOT
			religion = ROOT
			adm = 5
			dip = 4
			mil = 6
			age = 34
			claim = 20
		}
		event_target:gecko_pol_nobles = {

	desc = gecko.8.d

			OR = {
				 has_estate_privilege = estate_gc_sipahi 
				 has_estate_privilege = estate_gc_noble_levies
			} 

	}
	picture = MAMLUK_WARRIORS_eventPicture

	is_triggered_only = yes

	option = {
		name = gecko.8.a
	}
	option = {
		name = gecko.8.b
		if = {
			limit = {
				has_estate = estate_nomadic_tribes
			}
			add_estate_loyalty = {
				estate = estate_nomadic_tribes
				loyalty = -30
			}
			capital_scope = { 
				spawn_rebels = {
						type = tribal_rebels
						size = 1
					}
			}
			remove_estate_privilege = estate_gc_noble_levies
			remove_estate_privilege = estate_gc_sipahi
		}
		if = {
			limit = {
				has_estate = estate_nobles
			}
			add_estate_loyalty = {
				estate = estate_nobles
				loyalty = -30
			}
			capital_scope = { 
				spawn_rebels = {
						type = noble_rebels
						size = 1
					}
			}
			remove_estate_privilege = estate_gc_noble_levies
			remove_estate_privilege = estate_gc_sipahi
		}		
	}
}
